// 交集数组修改
  var A = [1,2,4]
  var B = [1,2,5,6]
  
  B = B.map(b=>{
     let w = ''
     for(let a of A){
        if(a === b)
        {
          w = 'x';
          break;
        }
        else{
          w = 'y'
        }
     }  
     
     return w;
  })
  
  
  // 并集,交集,差集
  
  const arr1 = [1, 2, 3],
      arr2 = [3, 4, 5];

//数组并集
console.log([...new Set([...arr1, ...arr2])]);
//输出：[1, 2, 3, 4, 5]

//数组交集 保留 a 和 b 中都有的
console.log(arr1.filter((v) => new Set(arr2).has(v)));
//输出：[3]

//数组差集
console.log(arr1.filter((v) => !new Set(arr2).has(v)));
//输出：[1, 2]
console.log(arr2.filter((v) => !new Set(arr1).has(v)));
//输出：[4, 5]
